scripted_gui = {
	wsh_congress_button_container = {
		window_name = "congress_button_container"
		context_type = player_context
		parent_window_token = politics_tab
		visible = { is_ai = no tag = WSH }
		effects = {
			congress_button_click = {
				if = {
					limit = { tag = WSH }
					WSH_recalculate_all_seats = yes
				}
				if = {
					limit = {
						NOT = { has_country_flag = congress_visible }
					}
					set_country_flag = congress_visible
				}
				else = {
					clr_country_flag = congress_visible
					clr_country_flag = congress_roster_visible
				}
			}
		}
	}
	wsh_congress_screen_container = {
		window_name = "congress_gui"
		context_type = player_context
		parent_window_token = politics_tab
		visible = {
			is_ai = no
			has_country_flag = congress_visible
			tag = WSH
		}
		dynamic_lists = {
			wsh_senate_diagram_container = {
				array = wsh_senate_gui_x
				change_scope = no
				entry_container = wsh_senate_gui_seat
				index = seat_idx
			}
			wsh_house_diagram_container = {
				array = wsh_house_gui_x
				change_scope = no
				entry_container = wsh_house_gui_seat
				index = house_idx
			}
		}
		properties = {
			wsh_senate_seat_icon = {
				x = wsh_senate_gui_x^seat_idx
				y = wsh_senate_gui_y^seat_idx
				frame = wsh_senate_gui_frame^seat_idx
			}
			wsh_house_seat_icon = {
				x = wsh_house_gui_x^house_idx
				y = wsh_house_gui_y^house_idx
				frame = wsh_house_gui_frame^house_idx
			}
			toggle_names_button = { frame = usa_show_names_frame }
		}
		effects = {
			toggle_names_button_click = {
				if = {
					limit = { has_country_flag = usa_show_congress_names }
					clr_country_flag = usa_show_congress_names
					set_variable = { usa_show_names_frame = 1 }
				}
				else = {
					set_country_flag = usa_show_congress_names
					set_variable = { usa_show_names_frame = 2 }
				}
			}
			congress_close_button_click = {
				clr_country_flag = congress_visible
				clr_country_flag = congress_roster_visible
			}
			open_roster_button_click = {
				if = {
					limit = { has_country_flag = congress_roster_visible }
					clr_country_flag = congress_roster_visible
				}
				else = { set_country_flag = congress_roster_visible }
			}
		}
	}
	wsh_congress_roster_container = {
		window_name = "wsh_congress_roster_panel"
		context_type = player_context
		parent_window_token = politics_tab
		visible = {
			is_ai = no
			has_country_flag = congress_visible
			has_country_flag = congress_roster_visible
			tag = WSH
		}
		dynamic_lists = {
			wsh_roster_grid = {
				array = wsh_roster_tokens
				change_scope = no
				entry_container = wsh_roster_entry
				index = roster_idx
			}
		}
		properties = {
			roster_badge_icon = { frame = wsh_roster_frames^roster_idx }
			roster_color_icon = { frame = wsh_roster_colors^roster_idx }
		}
	}
}
