WSH_debug_activation = {
	wsh_toggle_debug_menu = {
		icon = generic_prepare_civil_war
		available = { always = yes }
		fire_only_once = no
		cost = 0
		complete_effect = {
			if = {
				limit = { has_country_flag = wsh_backend_debug_active }
				clr_country_flag = wsh_backend_debug_active
			}
			else = { set_country_flag = wsh_backend_debug_active }
		}
	}
}
