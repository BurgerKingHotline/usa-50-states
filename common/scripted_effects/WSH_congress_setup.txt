			set_temp_variable = { wsh_eval_val = wsh_temp_total_seats^wsh_curr_idx }
			if = {
				limit = {
					check_variable = { wsh_eval_val > wsh_max_val }
				}
				set_temp_variable = { wsh_max_val = wsh_eval_val }
				set_temp_variable = { wsh_max_idx = wsh_curr_idx }
			}
			add_to_temp_variable = { wsh_curr_idx = 1 }
		}
		# Push the properties of the highest found party into the final arrays
		add_to_array = { wsh_roster_tokens = wsh_temp_tokens^wsh_max_idx }
		add_to_array = { wsh_roster_name_idx = wsh_temp_name_idx^wsh_max_idx }
		add_to_array = { wsh_roster_frames = wsh_temp_frames^wsh_max_idx }
		add_to_array = { wsh_roster_colors = wsh_temp_colors^wsh_max_idx }
		add_to_array = { wsh_roster_senate_seats = wsh_temp_senate_seats^wsh_max_idx }
		add_to_array = { wsh_roster_house_seats = wsh_temp_house_seats^wsh_max_idx }
		# Set its total value to -2 so it is ignored on the next iteration of the outer loop
		set_variable = { wsh_temp_total_seats^wsh_max_idx = -2 }
		# Decrement items left to sort
		subtract_from_temp_variable = { wsh_items_to_sort = 1 }
	}
}
